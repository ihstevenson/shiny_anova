# shiny_anova
shinyR demo for one-way ANOVA

Ian Stevenson, CC-BY, 2015

# Installation and Usage - no clone needed
    install.packages("shiny")

# Installation and Usage - if you clone the repo
    install.packages("shiny")
    library("shiny")
    runApp("shiny_anova")
